\section{Overview}

This thesis studies \emph{mean-field games with common noise}: stochastic
differential games in which a continuum of agents interact through a shared
empirical distribution, and where all agents are simultaneously exposed to a
common Brownian motion in addition to their individual idiosyncratic noise.
The central mathematical objects are:
\begin{itemize}
    \item the \emph{conditional measure flow} $t \mapsto \mu_t = \mathcal{L}(X_t
    \mid \mathcal{F}_t^{W^0})$, the conditional distribution of the representative
    agent's state given the common-noise filtration;
    \item the \emph{master equation}, a nonlinear second-order PDE on the space of
    probability measures $\mathcal{P}_2(\mathbb{R}^d)$ governing the value function
    $U(t, x, \mu)$ of the game; and
    \item the \emph{FBSDE characterisation}, a system of coupled
    forward-backward stochastic differential equations (FBSDEs) whose solution
    characterises the Nash equilibrium through the Stochastic Maximum Principle.
\end{itemize}

The thesis makes four contributions: a quantitative contraction theorem for the
FBSDE fixed-point map (Chapter 3); a convergent and stable numerical scheme for
the coupled FBSDE--Fokker--Planck system (Chapter 4); a rigorous particle-method
analysis including propagation-of-chaos estimates and computational complexity
(Chapter 5); and a partial formal verification of the core estimates in Lean 4
(Chapter 6). Throughout, the canonical linear-quadratic MFG serves as the
benchmark, with a corrected master-equation derivation (Appendix A) providing the
closed-form reference solution.

\section{What Is a Mean-Field Game?}

\subsection{The $N$-Player Stochastic Differential Game}

Consider $N$ agents indexed $i = 1, \ldots, N$, each with state process
$X_t^i \in \mathbb{R}^d$ satisfying
\begin{equation}
    dX_t^i = b(t, X_t^i, \mu_t^N, \alpha_t^i)\, dt + \sigma\, dW_t^i + \sigma_0\,
    dW_t^0, \qquad X_0^i \sim \mu_0,
    \label{eq:N-agent-sde}
\end{equation}
where $W^1, \ldots, W^N$ are independent Brownian motions (idiosyncratic noise),
$W^0$ is a common Brownian motion shared by all agents, and
$\mu_t^N = \frac{1}{N}\sum_{j=1}^N \delta_{X_t^j}$ is the empirical measure.
Each agent $i$ chooses a control $\alpha_t^i$ to minimise its individual cost
\[
    J^i(\alpha^i; \alpha^{-i}) = \mathbb{E}\left[ \int_0^T f(t, X_t^i, \mu_t^N,
    \alpha_t^i)\, dt + g(X_T^i, \mu_T^N) \right].
\]
The Nash equilibrium $(\alpha^{1,*}, \ldots, \alpha^{N,*})$ satisfies
$J^i(\alpha^{i,*}; \alpha^{-i,*}) \le J^i(\alpha^i; \alpha^{-i,*})$ for all
admissible $\alpha^i$ and all $i$.

\subsection{The Mean-Field Limit}

As $N \to \infty$, under regularity and monotonicity conditions on $b, f, g$,
the empirical interaction $\mu_t^N \to \mu_t$ in a suitable sense, and the
optimal control $\alpha^{i,*} \to \alpha^*(t, X_t^i, \mu_t)$ for a \emph{common}
feedback function $\alpha^*$. The limit is described by a McKean--Vlasov SDE:
\begin{equation}
    dX_t = b(t, X_t, \mu_t, \alpha^*(t, X_t, \mu_t))\, dt + \sigma\, dW_t +
    \sigma_0\, dW_t^0, \qquad X_0 \sim \mu_0,
    \label{eq:mckean-vlasov}
\end{equation}
where $\mu_t = \mathcal{L}(X_t \mid \mathcal{F}_t^{W^0})$.

\textbf{Key structural observation.} In the presence of common noise, the conditional
measure $\mu_t$ is itself a stochastic process adapted to the common-noise
filtration $\mathcal{F}_t^{W^0}$. Individual states $X_t^i$ are adapted to the
enlarged filtration generated by both $W^0$ and $W^i$, but the equilibrium object
$\mu_t$ is measurable with respect to $W^0$ alone. This asymmetry distinguishes the
common-noise theory from the classical (idiosyncratic-only) MFG, where $\mu_t$ is
deterministic.

\subsection{The $\varepsilon$-Nash Property}

The mean-field equilibrium is justified by the following approximation result,
established rigorously in Chapter 3:

\begin{theorem}[$\varepsilon$-Nash Property, informal]
Let $\alpha^*$ be the mean-field equilibrium control. For any $\varepsilon > 0$,
there exists $N_0$ such that for all $N \ge N_0$:
\[
    J^i(\alpha^{*}; \alpha^{*,-i}) \le \inf_{\alpha^i} J^i(\alpha^i; \alpha^{*,-i})
    + \varepsilon,
\]
where $\alpha^{*,-i}$ means all agents except $i$ use the mean-field control.
\end{theorem}

This justifies solving the computationally tractable mean-field problem in place of
the intractable $N$-player game.

\section{The Common-Noise FBSDE Characterisation}

The mean-field equilibrium is characterised via the Stochastic Maximum Principle.
Let $H(t,x,\mu,p,\alpha) = p \cdot b(t,x,\mu,\alpha) - f(t,x,\mu,\alpha)$ be the
Hamiltonian and $H^*(t,x,\mu,p) = \sup_\alpha H(t,x,\mu,p,\alpha)$ the maximised
Hamiltonian. The equilibrium $(X^*, \mu^*, \alpha^*, p, q, r)$ satisfies:
\begin{align}
    dX_t^* &= b(t, X_t^*, \mu_t^*, \alpha_t^*)\, dt + \sigma\, dW_t + \sigma_0\,
        dW_t^0, & X_0^* &\sim \mu_0, \label{eq:forward-sde} \\
    dp_t &= -H_x(t, X_t^*, \mu_t^*, p_t)\, dt + q_t\, dW_t + r_t\, dW_t^0,
        & p_T &= -g_x(X_T^*, \mu_T^*), \label{eq:adjoint-bsde}\\
    \alpha_t^* &= \arg\max_\alpha H(t, X_t^*, \mu_t^*, p_t, \alpha),
        \label{eq:optimality-condition} \\
    \mu_t^* &= \mathcal{L}(X_t^* \mid \mathcal{F}_t^{W^0}).
        \label{eq:consistency-condition}
\end{align}
The four equations \eqref{eq:forward-sde}--\eqref{eq:consistency-condition} form
the FBSDE system studied in Chapter 3. The forward equation \eqref{eq:forward-sde}
is driven by the common noise; the backward equation \eqref{eq:adjoint-bsde} is a
\emph{stochastic} BSDE because $\mu_t^*$ is stochastic. The consistency condition
\eqref{eq:consistency-condition} couples the forward and backward equations through
the distribution of the state.

\section{The Master Equation}

The master equation is the fundamental PDE governing the value function
$U: [0,T] \times \mathbb{R}^d \times \mathcal{P}_2(\mathbb{R}^d) \to \mathbb{R}$
of the mean-field game. It reads
\begin{equation}
    -\partial_t U = H^*(t,x,\mu, D_x U) + \bar{b}(t,\mu) \cdot D_\mu U
    + \frac{\sigma^2 + \sigma_0^2}{2}\Delta_x U
    + \sigma_0^2 \,\mathrm{tr}(D_x D_\mu U)
    + \frac{\sigma_0^2}{2} \int D_\mu^2 U(\tilde{x},\tilde{x})\, \mu(d\tilde{x}),
    \label{eq:master-equation}
\end{equation}
with terminal condition $U(T,x,\mu) = g(x,\mu)$, where $D_\mu U$ denotes the
Lions derivative of $U$ with respect to $\mu$, and $\bar{b}(t,\mu)$ is the
aggregate drift of the equilibrium measure flow. The terms in
\eqref{eq:master-equation} have the following roles:
\begin{itemize}
    \item $H^* + \bar{b} \cdot D_\mu U$: individual optimisation and mean-field
    coupling;
    \item $\frac{\sigma^2+\sigma_0^2}{2}\Delta_x U$: smoothing from the total noise;
    \item $\sigma_0^2 \,\mathrm{tr}(D_x D_\mu U) + \frac{\sigma_0^2}{2}\int
    D_\mu^2 U\, d\mu$: It\^{o} correction terms from the stochasticity of $\mu_t$
    itself, present only when $\sigma_0 > 0$.
\end{itemize}
The common-noise terms (last two) are absent from the classical deterministic-measure
MFG master equation and constitute the primary analytic challenge of this work.

\section{State of the Art}

\subsection{Classical Mean-Field Games}

Mean-field games were introduced independently by Lasry and Lions
\citep{lasrylions2007} and by Huang, Caines, and Malham\'e \citep{huangetal2006}.
Both papers established the fundamental $N \to \infty$ limit, the PDE system (HJB
coupled with Fokker--Planck), and the $\varepsilon$-Nash property. Cardaliaguet
\citep{cardaliaguet2010notes} gave a definitive PDE treatment of the master equation
for the deterministic-measure case. The probabilistic theory was developed
comprehensively by Carmona and Delarue \citep{carmonadelarue2018}, establishing
well-posedness under monotonicity conditions via FBSDEs.

\subsection{Common-Noise Mean-Field Games}

The extension to common noise was initiated by Carmona, Delarue, and Lachapelle
\citep{carmona2013probabilistic}, who introduced the conditional measure flow
$\mu_t = \mathcal{L}(X_t \mid \mathcal{F}_t^{W^0})$ as the correct state variable.
Ahuja, Ren, and Yang \citep{ahujarenyang2019} established well-posedness under
displacement monotonicity without the strong convexity assumptions of earlier work.
The master equation \eqref{eq:master-equation} with common-noise terms was
analysed by Cardaliaguet, Delarue, Lasry, and Lions \citep{cardaliaguetetal2019}.
SPDE regularity for common-noise MFGs was established by Cardaliaguet, Seeger, and
Souganidis \citep{cardaliaguetseegersouganidis2025}.

\subsection{Numerical Methods}

Numerical methods for MFGs (without common noise) include the finite-difference
schemes of Achdou and Capuzzo-Dolcetta \citep{achdoucapuzzo2010}, the monotone
schemes of Carlini and Silva, and the machine-learning approaches of Carmona and
Lauri\`{e}re \citep{carmonalauriere2022}. For common-noise MFGs, the particle
approximation approach is natural (the empirical measure approximates $\mu_t^N$
directly), but convergence proofs require control of the interaction between the
particle system and the common noise. Chapter 4 develops a convergent combined
scheme and Chapter 5 analyses the particle approximation theory in depth.

\subsection{Propagation of Chaos}

The classical propagation-of-chaos theory is due to McKean \citep{mckean1966}
and Sznitman \citep{sznitman1991}. For McKean--Vlasov SDEs, quantitative
convergence rates in Wasserstein distance were established by Fournier and Guillin
\citep{fournierguillin2015}. The common-noise setting requires conditional chaos
estimates -- bounding $W_2(\mu_t^N, \mu_t)$ given $W^0$ -- which are developed
in Chapter 5.

\section{The Three Research Questions}

This thesis addresses three interconnected questions:

\paragraph{Q1 (Well-posedness).} Under what conditions on the Hamiltonian, the
coupling through the distribution, and the noise coefficients does the common-noise
MFG equilibrium exist uniquely? What are the explicit parameter-dependent contraction
constants, and how do they depend on $\sigma_0$?

\paragraph{Q2 (Numerics).} What is the optimal convergence rate of numerical schemes
for the common-noise FBSDE--Fokker--Planck system? How does the presence of common
noise affect the CFL stability condition, and what is the computational complexity?

\paragraph{Q3 (Particle approximation).} How rapidly does the $N$-particle system
converge to the mean-field limit, uniformly in time? Can variance-reduction
techniques improve the $O(N^{-1/2})$ Fournier--Guillin rate for smooth test
functions? What is the computational cost of a parallel particle implementation?

\section{Principal Contributions}

The principal mathematical contributions of this thesis are:

\begin{theorem}[Quantitative Well-posedness, Chapter 3]
\label{thm:intro-wellposedness}
Under Assumptions A1--A5 (Lipschitz, monotone, bounded coefficients), the
fixed-point map $\Phi: \mathcal{C}([0,T]; \mathcal{P}_2(\mathbb{R}^d)) \to
\mathcal{C}([0,T]; \mathcal{P}_2(\mathbb{R}^d))$ characterising the common-noise
MFG equilibrium is a contraction in the weighted Wasserstein distance
$d_\beta(\mu, \nu) = \sup_t e^{-\beta t} W_2(\mu_t, \nu_t)$ with explicit
contraction constant
\[
    \rho = \frac{C_\alpha^2 \cdot C_{\mathrm{BSDE}}}{2\beta} < 1,
\]
for $\beta$ large enough relative to the Lipschitz constants. Consequently, the
equilibrium $(X^*, \mu^*, \alpha^*, p, q, r)$ exists and is unique.
\end{theorem}

\begin{theorem}[Numerical Convergence, Chapter 4]
\label{thm:intro-numerical}
The combined Euler--Maruyama / particle / finite-difference scheme for
\eqref{eq:forward-sde}--\eqref{eq:consistency-condition} converges with weak error
\[
    \mathbb{E}[W_2(\mu_T^\Delta, \mu_T^*)] \le C\left(\Delta t^{1/2} + \Delta x
    + M^{-1/2}\right),
\]
where $C$ depends only on the model parameters and horizon $T$, under the CFL
condition $\Delta t = O(\Delta x^2)$.
\end{theorem}

\begin{theorem}[Propagation of Chaos, Chapter 5]
\label{thm:intro-chaos}
Let $(\mu_t^N)_{t \in [0,T]}$ be the empirical measure of the $N$-particle
interacting system. Under dissipativity Assumption D1,
\[
    \sup_{t \in [0,T]} \mathbb{E}\big[W_2(\mu_t^N, \mu_t^*)^2 \mid W^0\big]
    \le \frac{C}{N},
\]
uniformly over common-noise realisations $W^0$, where $C$ depends on the
coefficients, $T$, $d$, and the initial variance $\mathbb{E}[|X_0|^2]$ but not on $N$.
\end{theorem}

\begin{proposition}[Corrected LQ Master Equation, Appendix A]
The linear-quadratic common-noise MFG (Example~\ref{ex:lq-mfg}) admits the
explicit value function $V(t,x,\mu) = -\tfrac12 A_t x^2 - B_t x\bar\mu
- \tfrac12 C_t \bar\mu^2 - D_t x - E_t\bar\mu - F_t$ where $A_t, \ldots, F_t$
satisfy the corrected Riccati system derived in Appendix~A. The corrected system
differs from earlier derivations in (i) the signs of the $A_t^2/\lambda$ and $c_1$
terms in $\dot{A}_t$, and (ii) the Lions-derivative coupling terms in
$\dot{B}_t$--$\dot{F}_t$.
\end{proposition}

\section{The Canonical Linear-Quadratic Benchmark}

Throughout the thesis, the following linear-quadratic model serves as the canonical
benchmark against which numerical schemes are validated.

\begin{example}[LQ-MFG, canonical benchmark]
\label{ex:lq-mfg}
\begin{align*}
    \text{State:} & \quad dX_t = \big[\kappa(\bar\mu_t - X_t) + \alpha_t\big] dt
        + \sigma\, dW_t + \sigma_0\, dW_t^0, \\
    \text{Running cost:} & \quad f(t,x,\mu,\alpha) = \tfrac{c_1}{2}x^2 +
        \tfrac{\lambda}{2}\alpha^2, \\
    \text{Terminal cost:} & \quad g(x,\mu) = \tfrac{c_2}{2}(x-\bar\mu)^2,
\end{align*}
where $\bar\mu_t = \int x\, \mu_t(dx)$ is the population mean, $\kappa > 0$ is
the mean-reversion rate, $\lambda > 0$ is the control cost, and $c_1, c_2 > 0$
are running and terminal state costs. Baseline parameters: $\kappa=0.5$,
$\lambda=1$, $c_1=2$, $c_2=5$, $\sigma=0.2$, $\sigma_0=0.3$, $T=1$.
\end{example}

This model arises in many applications -- coordination games, interacting oscillators,
collective motion -- and is one of the few MFG models with a fully explicit
equilibrium. It satisfies all standing assumptions (A1--A5, D1) and thus serves as
a rigorous benchmark for both the theoretical estimates and the numerical schemes.

\section{Thesis Roadmap}

\paragraph{Chapter 2} reviews the mathematical prerequisites: Wasserstein geometry,
It\^{o} calculus with common noise, Sobolev spaces and SPDEs, Lions derivatives, and
the Stochastic Maximum Principle.

\paragraph{Chapter 3} establishes the quantitative well-posedness of the common-noise
MFG (Theorem~\ref{thm:intro-wellposedness}), including the contraction estimate and
the $\varepsilon$-Nash property.

\paragraph{Chapter 4} develops numerical approximation: discretisation of the FBSDE
system, particle approximation of the measure flow, finite-difference approximation
of the Fokker--Planck SPDE, combined error estimates, and CFL analysis. All schemes
are validated against the LQ benchmark.

\paragraph{Chapter 5} analyses particle methods in depth: the $N$-particle
McKean--Vlasov system, conditional propagation of chaos, empirical measure
convergence rates, variance-reduction techniques, and parallel / GPU implementation.

\paragraph{Chapter 6} presents the Lean 4 formal verification of key estimates
(Gronwall, contraction factor, LQ optimality, Wasserstein-2 Gaussian formula).

\paragraph{Chapter 7} discusses future directions: jump-diffusion extensions, the
social optimum (mean-field control), deep Galerkin solvers, and the route from
subsequential to full convergence via the Lata\l{}a--Oleszkiewicz inequality.

\paragraph{Appendix A} provides the full derivation and correction of the LQ
master equation, verified three independent ways.

\section{Key Notation and Standing Assumptions}

\textbf{Notation.}
$\mathcal{P}_2(\mathbb{R}^d)$ denotes the space of Borel probability measures on
$\mathbb{R}^d$ with finite second moment. $W_2(\mu,\nu)$ is the Wasserstein-2
distance. $\bar\mu = \int x\, \mu(dx)$ is the mean. $C([0,T];\mathcal{P}_2)$ is
the space of continuous measure-valued paths. All filtrations are assumed complete.
$\mathbb{E}_0[\cdot] = \mathbb{E}[\cdot \mid \mathcal{F}_T^{W^0}]$ denotes
conditional expectation given the full common-noise path.

\textbf{Assumptions A1--A5.}
\begin{itemize}
    \item[A1.] \textit{Lipschitz.} $b, f, g$ are jointly Lipschitz in $(x,\mu)$
    uniformly in $t$, with Lipschitz constant $L$.
    \item[A2.] \textit{Monotonicity.} $\langle b(t,x,\mu,\alpha) - b(t,x',\mu,\alpha),
    x-x'\rangle \le -\kappa_0|x-x'|^2$ for some $\kappa_0 > 0$ (drift
    dissipativity).
    \item[A3.] \textit{Uniform concavity.} $H(t,x,\mu,p,\cdot)$ is uniformly
    $\lambda_0$-concave in $\alpha$, with $\lambda_0 > 0$.
    \item[A4.] \textit{Monotone coupling.} $\mu \mapsto f(t,x,\mu,\alpha)$ and
    $\mu \mapsto g(x,\mu)$ satisfy the Lasry--Lions monotonicity condition.
    \item[A5.] \textit{Bounded volatility.} $\sigma, \sigma_0 > 0$ are constants.
\end{itemize}

\textbf{Assumption D1 (Dissipativity for chaos).}
$\langle b(t,x,\mu,\alpha^*(t,x,\mu)) - b(t,x',\mu,\alpha^*(t,x',\mu)), x-x'\rangle
\le -\kappa_1|x-x'|^2$ for some $\kappa_1 > 0$, uniformly in $t, \mu$.
